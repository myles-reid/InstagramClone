CREATE TABLE User {
	UserID INT PRIMARY KEY,
	Username NVARCHAR(50) NOT NULL,
	Email NVARCHAR(50) NOT NULL,
	Password NVARCHAR(50) NOT NULL,
	Biography NVARCHAR(500),
	ProfilePicture NVARCHAR(50),
	NumFollower INT,
	NumFollowing INT
}
